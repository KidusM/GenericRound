        [HttpPost]
        public IActionResult SelectDS(string dsIdSelected)
        {
            //HttpContext.Session.SetString("_SelectedDsGroup", "");
            //HttpContext.Session.SetString("_SelectedDsCountry", "");
            //HttpContext.Session.SetString("_SelectedDsCity", "");
            //HttpContext.Session.SetString("_SelectedDsSurveyType", "");
            //HttpContext.Session.SetString("_SelectedDsDSID", "");

            var dsInfo = _context.Surveys.Include(a => a.Ds)
                            .Where(x => x.Dsid == dsIdSelected && x.SurveyBegin <= todaysDate && x.SurveyEnd > todaysDate)
                            .Select(aa => new { aa.Dsid, aa.Ds.Country, aa.Ds.City, aa.Ds.Dsgroup, aa.SurveyBegin, aa.SurveyEnd, aa.SurveyType })
                            .FirstOrDefault();

            HttpContext.Session.SetString("_SelectedDsGroup", dsInfo.Dsgroup.ToString().Trim());
            HttpContext.Session.SetString("_SelectedDsCountry", dsInfo.Country.ToString().Trim());
            HttpContext.Session.SetString("_SelectedDsCity", dsInfo.City.ToString().Trim());
            HttpContext.Session.SetString("_SelectedDsSurveyType", dsInfo.SurveyType.ToString().Trim());
            HttpContext.Session.SetString("_SelectedDsDSID", dsIdSelected.ToString().Trim());


            //ViewBag.DSDetails = string.Format("Duty Station and Group:  {0}, {1}", dsIdSelected, dsGroup);
            return View();
        }
